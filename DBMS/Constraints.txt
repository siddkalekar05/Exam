-- Step 1: Create Database
CREATE DATABASE college_db;
USE college_db;

-- Step 2: Create Department Table (PRIMARY KEY + UNIQUE + NOT NULL)
CREATE TABLE department (
    dept_id INT PRIMARY KEY,
    dept_name VARCHAR(30) UNIQUE NOT NULL
);

-- Step 3: Create Student Table (ALL CONSTRAINTS)
CREATE TABLE student (
    student_id INT PRIMARY KEY,           -- PRIMARY KEY
    name VARCHAR(50) NOT NULL,            -- NOT NULL
    age INT CHECK (age >= 18),            -- CHECK
    email VARCHAR(50) UNIQUE,             -- UNIQUE
    dept_id INT,                          -- FOREIGN KEY column
    FOREIGN KEY (dept_id) REFERENCES department(dept_id)
);

-- Step 4: Additional Table for NOT NULL Example
CREATE TABLE employee (
    emp_id INT PRIMARY KEY,
    emp_name VARCHAR(50) NOT NULL,
    salary INT
);

-- Step 5: Additional Table for CHECK Constraint Example
CREATE TABLE account (
    acc_no INT PRIMARY KEY,
    balance INT CHECK (balance >= 1000)
);

-- Step 6: Additional Table for UNIQUE Constraint Example
CREATE TABLE users (
    user_id INT PRIMARY KEY,
    email VARCHAR(50) UNIQUE
);
