void setup(){
  Serial.begin(115200);
  pinMode(2, OUTPUT);
  Serial.println("Warming up sensor...");
  delay(10000);
}

void loop(){
  int gasvalue = analogRead(34);
  Serial.print("Gas Level: ");
  Serial.println(gasvalue);

  if (gasvalue > 1000){
    digitalWrite(2, HIGH);
    Serial.println(" !!! DANGER: Gas Leak Detected !!!");
  }
  else{
    digitalWrite(2, LOW);
    Serial.println("Status: Environment Safe");
  }
}
